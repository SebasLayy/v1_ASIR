# 2. Topologia fisica en GNS3

  • R1 (C7200) - Conexion troncal al Switch2 (g2/0.10 y .20), enlace a MikroTik (g0/0), enlace a R3 (g1/0).
  • R3 (C7200) - Conexion a R1 (g0/0), a R4 (g1/0) y red interna Red3 (g2/0) con PC3/VPCS.
  • R4 (C7200) - Conexion a R3 (g2/0), a MikroTik (f0/0) y LAN Red4 (f1/0) con PC4.
  • Switch2 (C3725) - Trunk f1/0 - R1 g2/0. Access ports: f1/1-f1/8 VLAN10 (Servidor, PC1). f1/9-f1/15 VLAN20 (Cliente0 Alpine).
  • MikroTik RouterOS - Ether0 (172.16.10.81/30) a R1, Ether1 (172.16.10.94/30) a R4, Ether2 conectado a red NAT VBox (IP fija .2/24).
  • Servidor Alpine - conectado a VLAN10 (IP fija .50, Apache + SSH).
  • Cliente0 Alpine - VLAN20, IP por DHCP (R1).
  • PC1, PC3, PC4 (VPCS) - simulan hosts finales.

# 3. Configuracion paso a paso (clave para exito)

# Switch Cisco 3725 (Switch2)

enable
configure terminal
vlan 10
 name RED1_VLAN10
vlan 20
 name RED2_VLAN20
interface range fastEthernet 1/1 - 8
 switchport mode access
 switchport access vlan 10
 spanning-tree portfast
interface range fastEthernet 1/9 - 15
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
interface fastEthernet 1/0
 switchport mode trunk
 switchport trunk allowed vlan 10,20
end
write memory
# Explicacion: Se crean dos VLANs (10 y 20). Los puertos del 1/1 al 1/8 pertenecen a la VLAN10 (servidor y PC1). Los puertos 1/9-15 a VLAN20 (cliente0). El puerto 1/0 es troncal hacia R1 y lleva ambas VLANs etiquetadas.

# Router R1 - DHCP + Subinterfaces + OSPF + default route

conf t
ip dhcp excluded-address 172.16.10.49 172.16.10.50
ip dhcp excluded-address 172.16.10.63
ip dhcp excluded-address 172.16.10.65 172.16.10.79
!
ip dhcp pool vlan10
 network 172.16.10.48 255.255.255.240
 default-router 172.16.10.49
 dns-server 8.8.8.8
!
ip dhcp pool vlan20
 network 172.16.10.64 255.255.255.240
 default-router 172.16.10.65
 dns-server 8.8.8.8
!
interface gigabitEthernet 2/0
 no ip address
!
interface gigabitEthernet 2/0.10
 encapsulation dot1Q 10
 ip address 172.16.10.49 255.255.255.240
!
interface gigabitEthernet 2/0.20
 encapsulation dot1Q 20
 ip address 172.16.10.65 255.255.255.240
!
interface gigabitEthernet 0/0
 ip address 172.16.10.82 255.255.255.252
 no shutdown
!
interface gigabitEthernet 1/0
 ip address 172.16.10.85 255.255.255.252
 no shutdown
!
router ospf 1
 router-id 10.10.10.1
 auto-cost reference-bandwidth 10000
 network 172.16.10.0 0.0.0.255 area 0
!
ip route 0.0.0.0 0.0.0.0 172.16.10.81
end
wr
# Explicacion: R1 actua como gateway para las VLANs 10 y 20 mediante subinterfaces. El servidor (.50) queda excluido del DHCP. Ademas, redistribuye el default route hacia MikroTik y corre OSPF para conocer todas las redes internas.

# MikroTik RouterOS (NAT, port forwarding, hairpin)

/ip address add address=172.16.10.81/30 interface=ether0
/ip address add address=172.16.10.94/30 interface=ether1
/ip address add address=192.168.10.2/24 interface=ether2
/ip route add gateway=192.168.10.1
/ip firewall nat add chain=srcnat out-interface=ether2 action=masquerade
/ip firewall nat add chain=dstnat dst-port=80 protocol=tcp in-interface=ether2 action=dst-nat to-addresses=172.16.10.50 to-ports=80
/ip firewall nat add chain=dstnat dst-port=2221 protocol=tcp in-interface=ether2 action=dst-nat to-addresses=172.16.10.50 to-ports=22
/ip firewall nat add chain=srcnat src-address=172.16.10.0/24 dst-address=172.16.10.50 protocol=tcp dst-port=80 out-interface=bridge action=masquerade
/ip firewall nat add chain=srcnat src-address=172.16.10.0/24 dst-address=172.16.10.50 protocol=tcp dst-port=22 out-interface=bridge action=masquerade
# Rutas estaticas hacia redes internas via Cisco
/ip route add dst-address=172.16.10.0/27 gateway=172.16.10.82
/ip route add dst-address=172.16.10.32/28 gateway=172.16.10.93
/ip route add dst-address=172.16.10.48/28 gateway=172.16.10.82
/ip route add dst-address=172.16.10.64/28 gateway=172.16.10.82
# Explicacion: Se configura NAT masquerade para salida a internet. Ademas, se redirige el puerto 80 (HTTP) y el 2221 (SSH) hacia el servidor interno .50. La regla hairpin permite que desde la LAN se pueda acceder a la IP publica 192.168.10.2.

# Servidor Alpine (Apache + SSH) - IP estatica

# /etc/network/interfaces
auto eth0
iface eth0 inet static
    address 172.16.10.50
    netmask 255.255.255.240
    gateway 172.16.10.49
    dns-nameservers 8.8.8.8
# Instalar servicios
apk update
apk add apache2 openssh
rc-service apache2 start
rc-update add apache2 default
echo "
Servidor Corporativo - GNS3
" > /var/www/localhost/htdocs/index.html

# 5. Puntos criticos para que todo funcione

  • # Verificar que el switch tiene los puertos en las VLANs correctas (acceso y trunk).
  • # Asegurar que el servidor tenga IP fija fuera del rango DHCP.
  • # Comprobar vecinos OSPF: show ip ospf neighbor en R1, R3, R4.
  • # La ruta por defecto en R1 apunta a MikroTik (.81). MikroTik apunta a VBox (.1).
  • # El firewall de MikroTik debe permitir el forward.
  • # Desde cualquier equipo interno hacer ping a 8.8.8.8 para validar salida a internet.
